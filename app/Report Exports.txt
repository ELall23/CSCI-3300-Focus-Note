import RNFS from 'react-native-fs';

const exportToMarkdown = async (notes, filename) => {
  try {
    let mdContent = "";
    notes.forEach(note => {
      mdContent += `## ${note.category}\n${note.content}\n\n`;
    });

    const filePath = `${RNFS.DocumentDirectoryPath}/${filename}`;
    await RNFS.writeFile(filePath, mdContent, 'utf8');
    console.log('File saved at:', filePath);
    return filePath;
  } catch (error) {
    console.error('Error exporting Markdown:', error);
    throw error;
  }
};
