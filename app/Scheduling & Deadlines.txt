import ical from 'ical-generator';

const createCalendarReminder = (documentTitle, deadline) => {
  const calendar = ical({ name: 'Deadlines' });
  calendar.createEvent({
    start: deadline,
    end: deadline,
    summary: `Deadline: ${documentTitle}`,
  });

  const filePath = `${RNFS.DocumentDirectoryPath}/${documentTitle}_deadline.ics`;
  RNFS.writeFile(filePath, calendar.toString(), 'utf8');
  return filePath;
};

// Usage:
const deadline = new Date(2024, 11, 31, 23, 59);
createCalendarReminder("Pentest_Report", deadline);
